            <input type="number" id="loan-rate" placeholder="Interest Rate (%)">
            <input type="number" id="loan-term" placeholder="Months">
            <button class="btn-action" onclick="calcLoan()">Calculate Monthly Payment</button>
            <div id="loan-result" style="margin-top: 20px; text-align: center; font-size: 1.5rem; color: var(--primary);"></div>

        <button class="nav-item" onclick="showTab('loan', this)">Loans</button>

    </div>

    <script>
        let data = JSON.parse(localStorage.getItem('biz_suite')) || { transactions: [], goal: 10000 };

        function showTab(tabId, el) {
            document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
            document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active-nav'));
            document.getElementById('tab-' + tabId).classList.add('active');
            el.classList.add('active-nav');
        }

        function addEntry() {
            const desc = document.getElementById('desc').value;
            const amt = parseFloat(document.getElementById('amount').value);
            const type = document.getElementById('type').value;
            if (!desc || isNaN(amt)) return;

            data.transactions.push({ desc, amt, type, date: new Date().toLocaleDateString() });
            localStorage.setItem('biz_suite', JSON.stringify(data));
            updateUI();
            document.getElementById('desc').value = '';
            document.getElementById('amount').value = '';
        }

        function calcLoan() {
            const p = parseFloat(document.getElementById('loan-amount').value);
            const r = parseFloat(document.getElementById('loan-rate').value) / 100 / 12;
            const n = parseFloat(document.getElementById('loan-term').value);
            
            const emi = (p * r * Math.pow(1 + r, n)) / (Math.pow(1 + r, n) - 1);
            document.getElementById('loan-result').innerHTML = `Monthly: $${emi.toFixed(2)}`;
        }

        function updateUI() {
            let income = 0, expense = 0;
            const list = document.getElementById('list');
            list.innerHTML = '';

            data.transactions.forEach(t => {
                if (t.type === 'income') income += t.amt; else expense += t.amt;
                list.innerHTML += `<div class="item"><span>${t.desc}</span> <span style="color:${t.type === 'income' ? '#22c55e' : '#ef4444'}">${t.type === 'income' ? '+' : '-'}$${t.amt}</span></div>`;
            });

            const profit = income - expense;
            document.getElementById('total-profit').innerText = `$${profit.toFixed(2)}`;
            
            // Update Goal Bar
            const pct = Math.min((profit / data.goal) * 100, 100);
            document.getElementById('goal-fill').style.width = pct + '%';
            document.getElementById('goal-pct').innerText = Math.floor(pct) + '%';
        }

        updateUI();
    </script>
